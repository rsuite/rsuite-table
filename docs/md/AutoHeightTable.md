const App = () => {
  const [size, setSize] = React.useState(fakeData.length);
  const [autoHeight, setAutoHeight] = React.useState(true);

  return (
    <div>
      <Stack spacing={10} divider={<Divider vertical />}>
        <span>
          <Checkbox
            checked={autoHeight}
            onChange={(_v, checked) => {
              setAutoHeight(checked);
            }}
          >
            autoHeight
          </Checkbox>
        </span>

        <span>
          Size:{' '}
          <Input
            type="text"
            style={{ width: 100, display: 'inline-block' }}
            onChange={setSize}
            value={size}
          />{' '}
          rem
        </span>
      </Stack>
      <hr />

      <Table
        height={400}
        autoHeight={autoHeight}
        cellBordered
